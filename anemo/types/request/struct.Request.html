pub fn set_timeout(&mut self, timeout: Duration)

Set the max duration the request is allowed to take.

The duration will be formatted as nanoseconds as an u64 in the TIMEOUT header.

Example:

use std::time::Duration;
use anemo::Request;

let mut request = Request::new(());

request.set_timeout(Duration::from_secs(30));

let value = request.headers().get("timeout").unwrap();

assert_eq!(
    value,
    &Duration::from_secs(30).as_nanos().to_string(),
);

pub fn timeout(&self) -> Option<Duration>

Returns the previously-set timeout on this Request, or None if not set or invalid.

Example:

use std::time::Duration;
use anemo::Request;

let request = Request::new(()).with_timeout(Duration::from_secs(30));
let timeout = request.timeout().unwrap();

assert_eq!(
    timeout,
    Duration::from_secs(30),
);
